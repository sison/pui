    var tree1 = new Tree({
        tree: tree.dom,     // 容器id
        data: startData.theme3,
        main: tree.main,        //  需要缩进的列
        width: tree.width,      //  列宽度
        alias: tree.alias,      //  列类名
        header: tree.header,        //  头部
        headersub: tree.headersub,      //  副头部
        collapse: tree.collapse,        //  列表树
        callback: function(self) {      // 渲染后回调
            toggleLevel(self);
            changePrecent(self);
            changeWeight(self);
        }
    });
    tree1.init();
