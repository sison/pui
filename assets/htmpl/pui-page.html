    totalItem: 111,     // 总条目
    pageSize: 10,       // 每页大小
    pageIndex: 1,       // 当前页
    singlePage: false,  // 此属性仅作用在单页翻页切换场景
    action: function(pi, ps) {
        ceng.msg('pageIndex： ' + pi + '，pageSize： ' + ps);
    }
});
